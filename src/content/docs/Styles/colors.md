## Color Names

```js
$('canvas').drawArc({
  fillStyle: 'darkblue',
  x: 50, y: 50,
  radius: 30
});
```

## Hex Values

```js
$('canvas').drawArc({
  fillStyle: '#009',
  x: 50, y: 50,
  radius: 30
});
```

## RGB Values

```js
$('canvas').drawArc({
  fillStyle: 'rgb(0, 0, 153)',
  x: 50, y: 50,
  radius: 30
});
```

## RGBA Values

```js
$('canvas').drawArc({
  fillStyle: 'rgba(0, 0, 153, 0.5)',
  x: 50, y: 50,
  radius: 30
});
```
